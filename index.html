02
api/leads.js — Vercel Serverless
import { google } from 'googleapis'; export default async function handler(req, res) { const auth = new google.auth.GoogleAuth({ credentials: { client_email: process.env.GOOGLE_CLIENT_EMAIL, private_key: process.env.GOOGLE_PRIVATE_KEY?.replace(/\\n/g, '\n'), }, scopes: ['https://www.googleapis.com/auth/spreadsheets'], }); const sheets = google.sheets({ version: 'v4', auth }); const { data } = await sheets.spreadsheets.values.get({ spreadsheetId: process.env.GOOGLE_SHEET_ID, range: 'Leads!A:N', }); const [headers, ...rows] = data.values; const leads = rows.map(r => Object.fromEntries(headers.map((h,i) => [h, r[i]||'']))); res.json({ leads, updatedAt: new Date().toISOString() }); }
